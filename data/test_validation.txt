# Ejemplos de AFDs para testing de las mejoras de validación

# AFD simple y completo
1:COMPLETE_AFD:q0,q1
2:COMPLETE_AFD:a,b
3:COMPLETE_AFD:q0
4:COMPLETE_AFD:q1
5:COMPLETE_AFD:q0,a,q1;q0,b,q0;q1,a,q1;q1,b,q0

# AFD incompleto (falta algunas transiciones)
1:INCOMPLETE_AFD:q0,q1,q2
2:INCOMPLETE_AFD:a,b
3:INCOMPLETE_AFD:q0
4:INCOMPLETE_AFD:q2
5:INCOMPLETE_AFD:q0,a,q1;q1,b,q2

# AFD para testing de merge
1:MERGE_TEST:s0,s1
2:MERGE_TEST:x,y
3:MERGE_TEST:s0
4:MERGE_TEST:s1
5:MERGE_TEST:s0,x,s1;s1,y,s0

# Continuación del mismo AFD (para testing de merge)
1:MERGE_TEST:s2
2:MERGE_TEST:z
4:MERGE_TEST:s2
5:MERGE_TEST:s0,z,s2;s2,x,s1
